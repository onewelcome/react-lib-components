The `FormTemplate` component will be used as a wrapper component that contains FormHeader, FormAside or Stepper and form. Its root element is the default HTML `div` tag and thus also accepts all of its HTML attributes. You should use this component whenever you want to create a `FormTemplate`.

# Examples

The `FromTemplate` component is a wrapper component for creating FormHeader, FormAside or Stepper and form. See the code snippet below that showcases how you can use it.

```jsx

const buttons: ReactElement<ButtonProps, typeof Button>[] = [
  <Button key="1" onClick={() => alert("Cancel button clicked.")} variant="text">
    Cancel
  </Button>,
  <Button key="2" disabled startIcon={<Icon icon={Icons.SaveOutline} />}>
    Save Draft
  </Button>,
  <Button key="3" onClick={() => alert("Submit button clicked.")} startIcon={<Icon icon={Icons.ReplyOutline} />}>
    Submit
  </Button>
];

 const formHeaderComponent =  <FormHeader
  title= {"Form Header Text"}
  buttons={buttons}>
    Form Header children
 </FormHeader>


 const initialStepperState: FormStepProps[] = [
  {
    status: "done",
    label: "Step 1",
    targetScrollViewId: "samlBaseInfo"
  },
  {
    status: "done",
    label: "Step 2",
    targetScrollViewId: "samlConnectionDetail"
  },
  {
    status: "current",
    label: "Step 3",
    targetScrollViewId: "samlFederationRequest"
  },
  {
    status: "waiting",
    label: "Step 4",
    targetScrollViewId: "samlAttributeMapping"
  }
];

const formAsideProps: Props = {
  direction: "vertical",
  textPosition: "bottom",
  steps: initialStepperState
};

const formAsideComponent = <FormAside {...formAsideProps}></FormAside>

<FormTemplate FormHeader={formHeaderComponent} FormAside={formAsideComponent}>
    Any form children
</FormTemplate>
```
